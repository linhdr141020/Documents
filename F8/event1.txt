     function Button({title,href,onClick}){
            let Component = 'button'
            let props = {}
            if(href){
                Component = 'a'
                props.href = href
            }
            if(onClick){
                props.onClick = onClick
            }
            return (
                <Component {...props}>{title}</Component>
            )
        }
    function App(){

            </div>
        )
    }
