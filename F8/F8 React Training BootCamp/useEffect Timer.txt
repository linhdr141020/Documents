import {useEffect,useState} from 'react'

function Content(){
    const [countdown,setCountdown] = useState(100)
    useEffect(()=>{
     const timer = setInterval(()=>{
            setCountdown(prev => prev - 1)
            console.log('countdown')
        },1000)
        return ()=> clearInterval(timer)
    },[])
    return (
        <div>
            <h1>{countdown}</h1>
        </div>
    )
}
export default Content
