import {useState} from 'react'

let gifts = [
  'CPU 19',
  'RAM 32GB RGB',
  'RGB Keyboard'
]
function App(){
  const [random, setRandom] = useState()

  let handleUpdate = () => {
   let i = Math.floor(Math.random() * gifts.length)
    console.log(gifts[i]);
    setRandom(gifts[i])
  }
  return (
    <div className='App' style={{padding: 20}}>
      <h1>{random || 'Chua co phan thuong'}</h1>
      <button onClick={handleUpdate}>Lay thuong</button>
    </div>
  )
}
export default App;
